# lpegnode

Master module for library.

```lua
local dot    =  require "lpegespalier/dot"
local elpatt =  require "lpegespalier/elpatt"
local Node   =  require "lpegespalier/node"
local Spec   =  require "lpegespalier/spec"


return { dot    =  dot,
       elpatt =  elpatt,
       node   =  Node,
       spec   =  Spec }
```
